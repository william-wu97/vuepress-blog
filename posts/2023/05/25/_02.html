# 第一步：引入组件注册组件

import DemoZou from '@/components/DemoZou.vue'
components: {
   DemoZou
},

# 第二步：使用组件

<DemoZou :list="obj"></DemoZou>

# 第四步：把一下代码复制到.vue的空文件里面(文件地址要跟第一步对应)

靠左

<template>
  <!-- 第一层 -->
  <div class="flex">
    <div
      v-for="(item, index) in obj"
      :class="{ heng: !(index == num - 1) && num }"
    >
      <div class="div" :class="{ xia: xia(item), shang: shang2 }">
        <!-- 可以写具体样式，都可以更改，如更改大小，需要定义一下下面的值 -->
        {{ item.name }}
      </div>
      <!-- <div class="flex heng">
        <div>
          <div class="div shang xia">二1</div>
          <div class="flex heng">
            <div>
              <div class="div shang">三1</div>
            </div>
            <div>
              <div class="div shang">三2</div>
            </div>
          </div>
        </div>
        <div>
          <div class="div shang">二2</div>
        </div>
      </div> -->
      <DemoZou
        v-if="item.children"
        :list="item.children"
        shang2="false"
        :num="item.children.length"
      ></DemoZou>
    </div>
  </div>
</template>
<script>
export default {
  name: 'DemoZou',
  props: ['list', 'shang2', 'num'],
  data() {
    return {
      obj: [],
      // 上划线
      shang: true
    }
  },
  created() {
    // 有值才会进来
    if (this.list) {
      this.obj = this.list
      this.shang = this.shang2
    }
  },
  computed: {
    // 计算属性计算下线
    xia(item) {
      return (item) => {
        if(!item.children){
            return ''
        }
        return item.children ? 'xia' : ''
      }
    }
  }
}
</script>
